\begin{frame}{title}
	\begin{align*}
		pV = {\rm constant.}
	\end{align*}
	
	\begin{align}
		L = \int^{V_2}_{V_1} pdV &= \frac{m}{M} RT \int^{V_2}_{V_1} \frac{dV}{V} \notag \\
		&= \frac{m}{M}RT \ln \frac{V_2}{V_1} \notag \\
		&= \frac{m}{M} RT \ln \frac{p_1}{p_2}
	\end{align}
	
	\begin{align}
		L = RT \ln \frac{V_2}{V_1} = RT \ln \frac{p_1}{p_2}
	\end{align}
\end{frame}
